¿Qué es un agente IA?

Un programa que actúa, no solo responde

A diferencia de un chatbot, un agente IA percibe su entorno (archivos, correos, pantalla), decide qué hacer y ejecuta acciones reales en tu computadora. Esta actividad te lleva desde cero hasta un agente funcional.

import ollama
import json
import re
from datetime import datetime
from rich.console import Console
from rich.panel import Panel
from rich.text import Text
from config import LLM_MODEL, LLM_TEMP, MODO_SEGURO
from memory.episodic import MemoriaEpisodica

console = Console()

# ── Prompt del sistema — define la personalidad del agente ──
# PERSONALIZA ESTO: cambia el nombre, especialidad y reglas
SYSTEM_PROMPT = """
Eres ARIA (Autonomous Reasoning Intelligence Agent), un agente IA personal
que corre en la computadora del usuario. Tu objetivo es ayudar a organizar
archivos, gestionar correo y automatizar tareas repetitivas.

HERRAMIENTAS DISPONIBLES:
Puedes usar las siguientes herramientas escribiendo exactamente este formato:

HERRAMIENTA: nombre_herramienta
PARAMETROS: {"param1": "valor1", "param2": "valor2"}
RAZON: por qué usas esta herramienta

Herramientas disponibles:
- listar_archivos: lista archivos de una carpeta. Params: {"ruta": "path"}
- organizar_carpeta: organiza archivos por tipo. Params: {"ruta": "path"}
- mover_archivo: mueve un archivo. Params: {"origen": "path", "destino": "path"}
- leer_correos: lee correos no leídos. Params: {"limite": 5}
- enviar_correo: envía un correo. Params: {"para": "email", "asunto": "txt", "cuerpo": "txt"}
- abrir_aplicacion: abre una app. Params: {"nombre": "nombre_app"}
- tomar_captura: captura la pantalla. Params: {}
- esperar: espera N segundos. Params: {"segundos": 3}

REGLAS IMPORTANTES:
1. Piensa paso a paso antes de actuar (muestra tu razonamiento)
2. Una herramienta a la vez, observa el resultado antes de continuar
3. Siempre explica al usuario lo que vas a hacer ANTES de hacerlo
4. Si una tarea puede causar pérdida de datos, pide confirmación
5. Cuando termines una tarea, di explícitamente "TAREA COMPLETADA"

Modo actual: {modo}
"""

class AgenteIA:
    """
    Agente IA que implementa el bucle ReAct:
    
    1. PERCIBIR: recibe la tarea del usuario
    2. RAZONAR:  el LLM decide qué herramienta usar y por qué
    3. ACTUAR:   ejecuta la herramienta
    4. OBSERVAR: registra el resultado
    5. REPETIR:  hasta completar la tarea o necesitar intervención humana
    """
    
    def __init__(self):
        self.historial = []
        self.herramientas = {}  # Se poblan en Fase 2
        self.memoria = MemoriaEpisodica()
        self.modo_seguro = MODO_SEGURO
        self.tareas_completadas = 0
        console.print(Panel("[bold green]🤖 ARIA — Agente IA Personal iniciado[/bold green]\n"
                            f"Modelo: {LLM_MODEL} | Modo seguro: {MODO_SEGURO}",
                            title="Sistema"))
    
    def registrar_herramienta(self, nombre: str, funcion, descripcion: str):
        """Añade una herramienta al repertorio del agente."""
        self.herramientas[nombre] = {"fn": funcion, "desc": descripcion}
        console.print(f"  [dim]+ Herramienta registrada: [cyan]{nombre}[/cyan][/dim]")
    
    def _llamar_llm(self, mensaje: str) -> str:
        """Llama al LLM y devuelve su respuesta en texto."""
        modo_txt = "SEGURO (describe pero no ejecuta)" if self.modo_seguro else "ACTIVO (ejecuta acciones reales)"
        sistema = SYSTEM_PROMPT.format(modo=modo_txt)
        
        self.historial.append({"role": "user", "content": mensaje})
        
        response = ollama.chat(
            model=LLM_MODEL,
            messages=[{"role": "system", "content": sistema}] + self.historial,
            options={"temperature": LLM_TEMP}
        )
        
        respuesta = response['message']['content']
        self.historial.append({"role": "assistant", "content": respuesta})
        return respuesta
    
    def _parsear_herramienta(self, texto: str) -> dict | None:
        """Extrae el nombre y parámetros de la herramienta del texto del LLM."""
        h_match = re.search(r'HERRAMIENTA:\s*(\w+)', texto)
        p_match = re.search(r'PARAMETROS:\s*(\{.*?\})', texto, re.DOTALL)
        
        if not h_match:
            return None
        
        nombre = h_match.group(1).strip()
        params = {}
        if p_match:
            try: params = json.loads(p_match.group(1))
            except: pass
        
        return {"herramienta": nombre, "params": params}
    
    def ejecutar_tarea(self, tarea: str, max_pasos: int = 8) -> str:
        """
        Bucle principal ReAct:
        Razona → Actúa → Observa → Repite hasta completar la tarea.
        """
        console.print(Panel(f"[bold yellow]📋 Nueva tarea:[/bold yellow] {tarea}"))
        self.memoria.registrar_tarea(tarea)
        
        contexto = tarea
        
        for paso in range(1, max_pasos + 1):
            console.print(f"\n[dim]Paso {paso}/{max_pasos}...[/dim]")
            
            # RAZONAR: el LLM decide qué hacer
            respuesta_llm = self._llamar_llm(contexto)
            console.print(Panel(respuesta_llm, title=f"[cyan]🧠 Razonamiento (paso {paso})[/cyan]", border_style="dim"))
            
            # ¿Tarea completada?
            if "TAREA COMPLETADA" in respuesta_llm.upper():
                self.tareas_completadas += 1
                self.memoria.marcar_completada(tarea)
                self._notificar(f"✅ Tarea completada: {tarea[:60]}")
                return respuesta_llm
            
            # ACTUAR: extraer y ejecutar herramienta
            llamada = self._parsear_herramienta(respuesta_llm)
            if not llamada:
                # No hay herramienta — es solo una respuesta conversacional
                break
            
            nombre_h = llamada["herramienta"]
            params   = llamada["params"]
            
            # OBSERVAR: ejecutar y capturar resultado
            if nombre_h in self.herramientas:
                if self.modo_seguro:
                    resultado = f"[MODO SEGURO] Simularía: {nombre_h}({params})"
                    console.print(f"  [yellow]🔒 Modo seguro: {resultado}[/yellow]")
                else:
                    try:
                        resultado = self.herramientas[nombre_h]["fn"](**params)
                        console.print(f"  [green]✓ {nombre_h}: {str(resultado)[:120]}[/green]")
                    except Exception as e:
                        resultado = f"Error: {e}"
                        console.print(f"  [red]✗ {nombre_h}: {resultado}[/red]")
            else:
                resultado = f"Herramienta '{nombre_h}' no disponible."
            
            # Pasar resultado como contexto para el siguiente paso
            contexto = f"Resultado de {nombre_h}: {resultado}\nContinúa con la tarea."
        
        return "Tarea procesada."
    
    def _notificar(self, mensaje: str):
        """Muestra notificación en terminal. En Fase 4 se agrega notificación del SO."""
        console.print(Panel(f"[bold green]{mensaje}[/bold green]", border_style="green"))
    
    def chat(self):
        """Modo interactivo: el usuario escribe tareas en la terminal."""
        console.print("\n[bold cyan]ARIA lista. Escribe una tarea o 'salir' para terminar.[/bold cyan]\n")
        while True:
            try:
                entrada = console.input("[bold yellow]▶ Tarea:[/bold yellow] ").strip()
            except (KeyboardInterrupt, EOFError):
                break
            if entrada.lower() in ['salir', 'exit', 'quit']:
                break
            if entrada.lower() == 'modo real':
                self.modo_seguro = False
                console.print("[red bold]⚠️ Modo REAL activado — el agente ejecutará acciones reales[/red bold]")
                continue
            if entrada:
                self.ejecutar_tarea(entrada)


if __name__ == '__main__':
    agente = AgenteIA()
    agente.chat()

import sqlite3
import json
from datetime import datetime
from config import MEMORY_DB


class MemoriaEpisodica:
    """
    Almacena cada tarea ejecutada como un "episodio":
    qué se pidió, qué herramientas se usaron, si tuvo éxito,
    cuánto tardó. Permite al agente aprender de su historial.
    
    Tablas:
        episodios  — cada tarea ejecutada
        herramientas_uso — frecuencia de uso de cada tool
        preferencias — patrones aprendidos del usuario
    """
    
    def __init__(self):
        self.db = str(MEMORY_DB)
        self._init_db()
    
    def _init_db(self):
        conn = sqlite3.connect(self.db)
        conn.executescript("""
            CREATE TABLE IF NOT EXISTS episodios (
                id          INTEGER PRIMARY KEY AUTOINCREMENT,
                timestamp   TEXT NOT NULL,
                tarea       TEXT NOT NULL,
                completada  INTEGER DEFAULT 0,
                n_pasos     INTEGER DEFAULT 0,
                duracion_s  REAL DEFAULT 0,
                herramientas_usadas TEXT DEFAULT '[]'
            );
            CREATE TABLE IF NOT EXISTS herramientas_uso (
                herramienta TEXT PRIMARY KEY,
                usos        INTEGER DEFAULT 0,
                exitosas    INTEGER DEFAULT 0,
                ultima_vez  TEXT
            );
            CREATE TABLE IF NOT EXISTS preferencias (
                clave       TEXT PRIMARY KEY,
                valor       TEXT,
                actualizado TEXT
            );
        """)
        conn.commit()
        conn.close()
    
    def registrar_tarea(self, tarea: str) -> int:
        """Inicia un episodio y devuelve su ID."""
        conn = sqlite3.connect(self.db)
        c = conn.cursor()
        c.execute("INSERT INTO episodios (timestamp, tarea) VALUES (?, ?)",
                  (datetime.now().isoformat(), tarea))
        eid = c.lastrowid
        conn.commit(); conn.close()
        return eid
    
    def marcar_completada(self, tarea: str, herramientas: list = None):
        conn = sqlite3.connect(self.db)
        conn.execute("""UPDATE episodios SET completada=1, herramientas_usadas=?
                       WHERE tarea=? AND completada=0""",
                     (json.dumps(herramientas or []), tarea))
        conn.commit(); conn.close()
    
    def registrar_uso_herramienta(self, herramienta: str, exito: bool):
        conn = sqlite3.connect(self.db)
        conn.execute("""INSERT INTO herramientas_uso (herramienta, usos, exitosas, ultima_vez)
                       VALUES (?, 1, ?, ?)
                       ON CONFLICT(herramienta) DO UPDATE SET
                           usos = usos + 1,
                           exitosas = exitosas + ?,
                           ultima_vez = ?""",
                     (herramienta, 1 if exito else 0, datetime.now().isoformat(),
                      1 if exito else 0, datetime.now().isoformat()))
        conn.commit(); conn.close()
    
    def tareas_similares(self, tarea: str, limite: int = 3) -> list:
        """Busca episodios anteriores similares a la tarea actual."""
        palabras = [p for p in tarea.lower().split() if len(p) > 3]
        if not palabras: return []
        conn = sqlite3.connect(self.db)
        c = conn.cursor()
        like = f"%{palabras[0]}%"
        c.execute("""SELECT tarea, completada, herramientas_usadas
                    FROM episodios WHERE tarea LIKE ? AND completada=1
                    ORDER BY timestamp DESC LIMIT ?""", (like, limite))
        resultados = c.fetchall()
        conn.close()
        return resultados
    
    def resumen(self) -> dict:
        conn = sqlite3.connect(self.db)
        c = conn.cursor()
        c.execute("SELECT COUNT(*), SUM(completada) FROM episodios")
        total, completadas = c.fetchone()
        c.execute("SELECT herramienta, usos FROM herramientas_uso ORDER BY usos DESC LIMIT 5")
        top_tools = c.fetchall()
        conn.close()
        return {"total_tareas": total or 0,
                "completadas": completadas or 0,
                "top_herramientas": top_tools}

import shutil
import os
from pathlib import Path
from datetime import datetime
from config import CARPETAS_GESTIONADAS, MAX_ARCHIVOS_POR_OP, CONFIRMAR_BORRADO

# ── Clasificación de archivos por extensión ───────────────
TIPOS_ARCHIVO = {
    "Imágenes":    ['.jpg', '.jpeg', '.png', '.gif', '.webp', '.svg', '.bmp'],
    "Documentos":  ['.pdf', '.docx', '.doc', '.txt', '.odt', '.rtf'],
    "Hojas":       ['.xlsx', '.xls', '.csv', '.ods'],
    "Presentaciones": ['.pptx', '.ppt', '.key'],
    "Código":      ['.py', '.js', '.ts', '.html', '.css', '.java', '.cpp', '.c'],
    "Videos":      ['.mp4', '.avi', '.mov', '.mkv', '.webm'],
    "Audio":       ['.mp3', '.wav', '.flac', '.ogg', '.m4a'],
    "Comprimidos": ['.zip', '.rar', '.7z', '.tar', '.gz'],
    "Ejecutables": ['.exe', '.msi', '.dmg', '.pkg', '.deb'],
}

def _tipo_archivo(extension: str) -> str:
    ext = extension.lower()
    for tipo, exts in TIPOS_ARCHIVO.items():
        if ext in exts: return tipo
    return "Otros"

def _ruta_segura(ruta: str) -> Path:
    """Verifica que la ruta esté dentro de las carpetas permitidas."""
    p = Path(ruta).resolve()
    if not any(p == c or c in p.parents for c in CARPETAS_GESTIONADAS):
        raise PermissionError(f"Ruta no permitida: {ruta}. Solo puedo acceder a carpetas configuradas.")
    return p


def listar_archivos(ruta: str = ".") -> str:
    """Lista archivos de una carpeta con información de tamaño y fecha."""
    p = Path(ruta).expanduser()
    if not p.exists(): return f"La ruta '{ruta}' no existe."
    
    archivos = [f for f in p.iterdir() if f.is_file()][:MAX_ARCHIVOS_POR_OP]
    carpetas = [d for d in p.iterdir() if d.is_dir()]
    
    resultado = [f"📁 {p} ({len(archivos)} archivos, {len(carpetas)} subcarpetas)\n"]
    for f in archivos:
        stat = f.stat()
        kb = stat.st_size / 1024
        fecha = datetime.fromtimestamp(stat.st_mtime).strftime("%d/%m/%Y")
        tipo = _tipo_archivo(f.suffix)
        resultado.append(f"  [{tipo:12}] {f.name:<40} {kb:>8.1f} KB  {fecha}")
    
    return "\n".join(resultado)


def organizar_carpeta(ruta: str) -> str:
    """
    Organiza todos los archivos de una carpeta en subcarpetas por tipo.
    Ejemplo: Downloads/ → Downloads/Imágenes/, Downloads/Documentos/, etc.
    """
    p = _ruta_segura(ruta)
    if not p.exists(): return f"Ruta no encontrada: {ruta}"
    
    movidos = []
    errores = []
    
    for archivo in p.iterdir():
        if not archivo.is_file(): continue
        if archivo.name.startswith('.'): continue  # Ignorar archivos ocultos
        
        tipo = _tipo_archivo(archivo.suffix)
        destino_dir = p / tipo
        destino_dir.mkdir(exist_ok=True)
        destino = destino_dir / archivo.name
        
        # Si ya existe, añadir timestamp al nombre
        if destino.exists():
            ts = datetime.now().strftime("%H%M%S")
            destino = destino_dir / f"{archivo.stem}_{ts}{archivo.suffix}"
        
        try:
            shutil.move(str(archivo), str(destino))
            movidos.append(f"{archivo.name} → {tipo}/")
        except Exception as e:
            errores.append(f"{archivo.name}: {e}")
    
    resumen = f"✅ Organizados: {len(movidos)} archivos\n"
    resumen += "\n".join(movidos[:10])
    if errores: resumen += f"\n⚠️ Errores: {len(errores)}"
    return resumen


def mover_archivo(origen: str, destino: str) -> str:
    """Mueve un archivo de origen a destino, ambos deben estar en rutas permitidas."""
    o = _ruta_segura(origen)
    d = Path(destino).expanduser()
    if not o.exists(): return f"Archivo no encontrado: {origen}"
    d.parent.mkdir(parents=True, exist_ok=True)
    shutil.move(str(o), str(d))
    return f"✅ Movido: {o.name} → {d}"

import schedule
import threading
import time
import sqlite3
import json
from datetime import datetime
from rich.console import Console
from config import MEMORY_DB

console = Console()


class ColaTareas:
    """
    Gestiona tareas programadas para el agente.
    
    Tipos de tarea:
    - inmediata: se ejecuta ahora
    - diaria: todos los días a una hora
    - intervalo: cada N minutos
    - unica: una sola vez en una fecha/hora específica
    
    Las tareas se guardan en SQLite para sobrevivir reinicios.
    """
    
    def __init__(self, agente):
        self.agente = agente
        self.db = str(MEMORY_DB)
        self._init_tabla()
        self.hilo_programador = None
    
    def _init_tabla(self):
        conn = sqlite3.connect(self.db)
        conn.execute("""
            CREATE TABLE IF NOT EXISTS tareas_programadas (
                id       INTEGER PRIMARY KEY AUTOINCREMENT,
                nombre   TEXT NOT NULL,
                descripcion TEXT,
                tipo     TEXT NOT NULL,
                config   TEXT,
                activa   INTEGER DEFAULT 1,
                ultima_ejecucion TEXT,
                proxima  TEXT
            )
        """)
        conn.commit(); conn.close()
    
    def agregar_tarea(self, nombre: str, descripcion: str,
                       tipo: str, config: dict = None) -> str:
        """
        Agrega una nueva tarea programada.
        
        tipo='diaria', config={'hora': '09:00', 'tarea': 'revisar correo'}
        tipo='intervalo', config={'minutos': 30, 'tarea': 'verificar descargas'}
        """
        conn = sqlite3.connect(self.db)
        conn.execute("""INSERT INTO tareas_programadas (nombre, descripcion, tipo, config)
                       VALUES (?, ?, ?, ?)""",
                     (nombre, descripcion, tipo, json.dumps(config or {})))
        conn.commit(); conn.close()
        
        self._programar(nombre, tipo, config)
        console.print(f"[green]📅 Tarea programada: [bold]{nombre}[/bold] ({tipo})[/green]")
        return f"Tarea '{nombre}' programada exitosamente."
    
    def _programar(self, nombre: str, tipo: str, config: dict):
        def ejecutar():
            console.print(f"\n[bold yellow]⏰ Ejecutando tarea programada: {nombre}[/bold yellow]")
            tarea_texto = config.get('tarea', nombre)
            resultado = self.agente.ejecutar_tarea(tarea_texto)
            # Actualizar última ejecución
            conn = sqlite3.connect(self.db)
            conn.execute("UPDATE tareas_programadas SET ultima_ejecucion=? WHERE nombre=?",
                        (datetime.now().isoformat(), nombre))
            conn.commit(); conn.close()
            return resultado
        
        if tipo == 'diaria':
            hora = config.get('hora', '09:00')
            schedule.every().day.at(hora).do(ejecutar).tag(nombre)
        elif tipo == 'intervalo':
            mins = config.get('minutos', 30)
            schedule.every(mins).minutes.do(ejecutar).tag(nombre)
    
    def iniciar_hilo(self):
        """Inicia el programador en un hilo separado para no bloquear el agente."""
        def loop():
            while True:
                schedule.run_pending()
                time.sleep(30)
        
        self.hilo_programador = threading.Thread(target=loop, daemon=True)
        self.hilo_programador.start()
        console.print("[dim]🕐 Programador de tareas iniciado en segundo plano[/dim]")
    
    def listar_tareas(self) -> str:
        conn = sqlite3.connect(self.db)
        c = conn.cursor()
        c.execute("SELECT nombre, tipo, ultima_ejecucion, activa FROM tareas_programadas")
        tareas = c.fetchall(); conn.close()
        if not tareas: return "No hay tareas programadas."
        resultado = "📅 Tareas programadas:\n"
        for t in tareas:
            estado = "✅" if t[3] else "⏸️"
            ultima = t[2][:16] if t[2] else "nunca"
            resultado += f"  {estado} {t[0]:<20} [{t[1]}] última: {ultima}\n"
        return resultado

def _decodificar_header(valor: str) -> str:
    """Decodifica cabeceras de correo que pueden venir en distintos encodings."""
    partes = decode_header(valor)
    resultado = []
    for parte, enc in partes:
        if isinstance(parte, bytes):
            resultado.append(parte.decode(enc or 'utf-8', errors='replace'))
        else:
            resultado.append(parte)
    return " ".join(resultado)


def leer_correos(limite: int = 5, carpeta: str = "INBOX") -> str:
    """
    Lee los últimos N correos no leídos de la bandeja especificada.
    Devuelve un resumen legible para el agente.
    """
    if not EMAIL_ADDRESS:
        return "Error: No hay cuenta de correo configurada en .env"
    
    try:
        mail = imaplib.IMAP4_SSL(IMAP_SERVER)
        mail.login(EMAIL_ADDRESS, EMAIL_PASSWORD)
        mail.select(carpeta)
        
        # Buscar correos no leídos
        _, ids = mail.search(None, 'UNSEEN')
        id_list = ids[0].split()
        
        if not id_list:
            mail.logout()
            return "📭 No hay correos no leídos."
        
        correos = []
        # Leer los más recientes (últimos N)
        for cid in id_list[-limite:]:
            _, data = mail.fetch(cid, '(RFC822)')
            msg = email.message_from_bytes(data[0][1])
            
            de     = _decodificar_header(msg.get('From', ''))
            asunto = _decodificar_header(msg.get('Subject', 'Sin asunto'))
            fecha  = msg.get('Date', '')[:25]
            
            # Extraer cuerpo del mensaje
            cuerpo = ""
            if msg.is_multipart():
                for parte in msg.walk():
                    if parte.get_content_type() == 'text/plain':
                        try: cuerpo = parte.get_payload(decode=True).decode('utf-8', errors='replace')[:300]
                        except: pass
                        break
            else:
                try: cuerpo = msg.get_payload(decode=True).decode('utf-8', errors='replace')[:300]
                except: pass
            
            correos.append(f"--- CORREO ---\nDe: {de}\nAsunto: {asunto}\nFecha: {fecha}\nCuerpo:\n{cuerpo}\n")
        
        mail.logout()
        return f"📬 {len(correos)} correos no leídos:\n\n" + "\n".join(correos)
    
    except Exception as e:
        return f"Error de correo: {e}"

    - Horarios habituales de uso (mañana/tarde/noche)
    - Herramientas más usadas por hora del día
    - Tipos de archivos que más gestiona
    - Tono preferido en correos
    - Tareas que se repiten (candidatas a automatización)
    
    Esto permite al agente:
    - Sugerir acciones proactivamente
    - Adaptar la verbosidad de sus respuestas
    - Pre-cargar herramientas frecuentes
    """
    
    def __init__(self):
        self.db = str(MEMORY_DB)
        self._init_tabla()
        self.cache = {}  # Cache en memoria para acceso rápido
    
    def _init_tabla(self):
        conn = sqlite3.connect(self.db)
        conn.executescript("""
            CREATE TABLE IF NOT EXISTS perfil (
                clave TEXT PRIMARY KEY,
                valor TEXT,
                tipo  TEXT DEFAULT 'string',
                actualizado TEXT
            );
            CREATE TABLE IF NOT EXISTS uso_horario (
                hora     INTEGER,
                herramienta TEXT,
                usos     INTEGER DEFAULT 1,
                PRIMARY KEY (hora, herramienta)
            );
        """)
        conn.commit(); conn.close()
    
    def registrar_uso(self, herramienta: str):
        """Registra el uso de una herramienta con la hora actual."""
        hora = datetime.now().hour
        conn = sqlite3.connect(self.db)
        conn.execute("""INSERT INTO uso_horario (hora, herramienta, usos)
                       VALUES (?, ?, 1)
                       ON CONFLICT(hora, herramienta)
                       DO UPDATE SET usos = usos + 1""", (hora, herramienta))
        conn.commit(); conn.close()
    
    def herramientas_recomendadas(self) -> list:
        """Devuelve las herramientas más usadas en la hora actual."""
        hora = datetime.now().hour
        conn = sqlite3.connect(self.db)
        c = conn.cursor()
        c.execute("""SELECT herramienta, usos FROM uso_horario
                    WHERE hora = ? ORDER BY usos DESC LIMIT 3""", (hora,))
        result = c.fetchall(); conn.close()
        return [r[0] for r in result]
    
    def guardar_preferencia(self, clave: str, valor, tipo: str = 'string'):
        """Guarda una preferencia aprendida del usuario."""
        val_str = json.dumps(valor) if tipo == 'json' else str(valor)
        conn = sqlite3.connect(self.db)
        conn.execute("""INSERT INTO perfil (clave, valor, tipo, actualizado)
                       VALUES (?, ?, ?, ?)
                       ON CONFLICT(clave) DO UPDATE SET
                           valor=?, actualizado=?""",
                     (clave, val_str, tipo, datetime.now().isoformat(),
                      val_str, datetime.now().isoformat()))
        conn.commit(); conn.close()
        self.cache[clave] = valor
    
    def obtener(self, clave: str, defecto=None):
        if clave in self.cache: return self.cache[clave]
        conn = sqlite3.connect(self.db)
        c = conn.cursor()
        c.execute("SELECT valor, tipo FROM perfil WHERE clave=?", (clave,))
        row = c.fetchone(); conn.close()
        if not row: return defecto
        val = json.loads(row[0]) if row[1] == 'json' else row[0]
        self.cache[clave] = val
        return val
    
    def detectar_patrones(self) -> str:
        """
        Analiza el historial y detecta tareas que se repiten.
        Candidatos a automatización.
        """
        conn = sqlite3.connect(self.db)
        c = conn.cursor()
        c.execute("""SELECT tarea FROM episodios
                    WHERE completada=1
                    ORDER BY timestamp DESC LIMIT 50""")
        tareas = [r[0] for r in c.fetchall()]; conn.close()
        
        if not tareas: return "No hay suficiente historial para detectar patrones."
        
        # Agrupar por primeras palabras (simplificación)
        prefijos = Counter()
        for t in tareas:
            palabras = t.lower().split()[:3]
            if palabras: prefijos[" ".join(palabras)] += 1
        
        repetidas = [(p, n) for p, n in prefijos.most_common(5) if n >= 2]
        if not repetidas: return "No se detectaron patrones repetitivos aún."
        
        resultado = "🔍 Patrones detectados (candidatos a automatizar):\n"
        for patron, veces in repetidas:
            resultado += f"  → '{patron}...' — {veces} veces\n"
        return resultado

def buscar_web(consulta: str) -> str:
    """
    Busca en DuckDuckGo y devuelve los primeros resultados.
    
    Tu desafío:
    1. Extraer el texto relevante de las páginas resultado
