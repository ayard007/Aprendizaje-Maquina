# ── CELDA 3: Exploración inicial del inventario ─────────────

def explorar_inventario(df):
    """
    Analiza el estado actual del inventario y genera un reporte
    de diagnóstico previo al cálculo de alertas.
    
    Parámetros:
        df (DataFrame): tabla de inventario
    Retorna:
        dict con métricas generales del inventario
    """
    
    # Calcular métricas generales
    total_productos   = len(df)
    
    # Productos donde el stock actual ya cayó por debajo del mínimo
    criticos = (df['stock_actual'] <= df['stock_minimo']).sum()
    
    # Valor total del inventario = suma de (stock * costo) de todos los productos
    valor_total = (df['stock_actual'] * df['costo_unitario']).sum()
    
    # Porcentaje de ocupación del almacén (stock actual / máximo * 100)
    df_analisis = df.assign(
        valor_producto = df['stock_actual'] * df['costo_unitario'],
        pct_ocupacion  = (df['stock_actual'] / df['stock_maximo'] * 100).round(1)
    )
    
    # ── Imprimir reporte de diagnóstico ──
    print("=" * 55)
    print("  DIAGNÓSTICO INICIAL DEL INVENTARIO")
    print("=" * 55)
    print(f"  Total de productos en catálogo : {total_productos}")
    print(f"  Productos en estado crítico    : {criticos}")
    print(f"  Valor total del inventario     : ${valor_total:,.2f}")
    print("─" * 55)
    
    # Mostrar los 3 productos con menor porcentaje de ocupación
    print("\n  Productos con MENOR nivel de stock:")
    mas_bajos = df_analisis.nsmallest(3, 'pct_ocupacion')
    for _, row in mas_bajos.iterrows():
        icono = "🔴" if row['stock_actual'] <= row['stock_minimo'] else "🟡"
        print(f"  {icono} {row['nombre']:22s} {row['pct_ocupacion']:5.1f}% lleno")
    
    print("=" * 55)
    
    return {'total':total_productos, 'criticos':criticos, 'valor':valor_total}

# Ejecutar la exploración
metricas = explorar_inventario(df)

Celda 4 — Fórmula del Punto de Reorden (ROP)
ROP = demanda durante el tiempo de entrega + stock de seguridad
PythonNumPyLógica
○
La fórmula del Punto de Reorden

El Punto de Reorden (ROP) es el nivel de inventario al que se debe lanzar una orden de compra para que el producto llegue justo antes de agotarse. La fórmula completa es:

ROP = (demanda_diaria × tiempo_entrega) + stock_seguridad

El stock de seguridad cubre la variabilidad: si la demanda sube inesperadamente o el proveedor tarda más. Se calcula como:

SS = Z × σ_demanda × √tiempo_entrega

Donde Z = 1.645 corresponde a un nivel de servicio del 95% (5% de probabilidad de quiebre). Este valor proviene de la distribución normal estándar. np.sqrt() calcula la raíz cuadrada.

# ── CELDA 4: Cálculo del Punto de Reorden (ROP) ─────────────

# Nivel de servicio del 95% → Z = 1.645 (valor de tabla normal estándar)
# Significa que el 95% del tiempo NO habrá quiebre de stock
Z = 1.645

def calcular_rop(df, z=1.645):
    """
    Calcula el Punto de Reorden y Stock de Seguridad para cada producto.
    
    Fórmulas:
        Stock Seguridad = Z × σ_demanda × √(tiempo_entrega)
        ROP = (demanda_diaria × tiempo_entrega) + Stock_Seguridad
        
    Parámetros:
        df : DataFrame del inventario
        z  : factor de nivel de servicio (default 1.645 = 95%)
    Retorna:
        DataFrame con columnas ROP y stock_seguridad añadidas
    """
    
    df_rop = df.copy()  # copia para no modificar el original
    
    # ── Stock de Seguridad (SS) ──────────────────────────────
    # Absorbe variaciones de demanda durante el tiempo de entrega
    # np.sqrt() aplica raíz cuadrada a toda la columna vectorialmente
    df_rop['stock_seguridad'] = (
        z
        * df_rop['desviacion_demanda']
        * np.sqrt(df_rop['tiempo_entrega_dias'])
    ).round(1)
    
    # ── Punto de Reorden (ROP) ──────────────────────────────
    # Demanda esperada durante el tiempo de entrega + colchón de seguridad
    df_rop['punto_reorden'] = (
        df_rop['demanda_diaria_promedio'] * df_rop['tiempo_entrega_dias']
        + df_rop['stock_seguridad']
    ).round(1)
    
    # ── Días de cobertura restantes ─────────────────────────
    # Cuántos días durará el stock actual al ritmo de consumo actual
    df_rop['dias_cobertura'] = (
        df_rop['stock_actual'] / df_rop['demanda_diaria_promedio']
    ).round(1)
    
    return df_rop

# Aplicar la función y actualizar nuestro DataFrame principal
df = calcular_rop(df, z=Z)

# Mostrar resultados de ROP para verificación
cols_ver = ['nombre','stock_actual','stock_seguridad','punto_reorden','dias_cobertura']
print("Puntos de Reorden calculados:\n")
print(df[cols_ver].to_string(index=False))

Celda 5 — Clasificación ABC del inventario

¿Qué es la clasificación ABC?

La clasificación ABC (Análisis de Pareto) agrupa productos según su impacto económico: Clase A (top 70% del valor total, control estricto), Clase B (siguiente 20%, control moderado), Clase C (último 10%, control básico). Permite enfocar los recursos de monitoreo donde más importa.

Usamos pd.cut() para segmentar un valor continuo (porcentaje acumulado) en categorías discretas (A, B, C) con rangos definidos.

# ── CELDA 5: Clasificación ABC ──────────────────────────────

def clasificar_abc(df):
    """
    Implementa la clasificación ABC (Pareto) del inventario.
    A = primero 70% del valor | B = siguiente 20% | C = último 10%
    """
    
    df_abc = df.copy()
    
    # Calcular el valor de cada producto en inventario
    df_abc['valor_inventario'] = df_abc['stock_actual'] * df_abc['costo_unitario']
    
    # Ordenar de mayor a menor valor (descendente)
    df_abc = df_abc.sort_values('valor_inventario', ascending=False)
    
    # Calcular porcentaje acumulado del valor total
    # cumsum() suma acumulada | div() divide por el total para obtener %
    total_valor = df_abc['valor_inventario'].sum()
    df_abc['pct_valor_acum'] = (
        df_abc['valor_inventario'].cumsum() / total_valor * 100
    )
    
    # pd.cut() asigna categoría según el rango del % acumulado
    # bins define los límites: 0-70% → A, 70-90% → B, 90-100% → C
    df_abc['clase_abc'] = pd.cut(
        df_abc['pct_valor_acum'],
        bins=[0, 70, 90, 100],
        labels=['A', 'B', 'C'],
        include_lowest=True
    )
    
    return df_abc

df = clasificar_abc(df)

# Resumen por clase
print("Clasificación ABC del inventario:\n")
resumen = df.groupby('clase_abc').agg(
    productos=('nombre','count'),
    valor_total=('valor_inventario','sum')
)
print(resumen)

¿Por qué calcular la fecha de agotamiento?

Saber que un producto tiene "6 unidades" no dice mucho. Saber que "se agotará en 7.5 días" sí genera urgencia. Al calcular la fecha estimada de agotamiento, el sistema puede ordenar las alertas por urgencia real, no solo por cantidad absoluta.

Usamos datetime.now() + timedelta(days=x) para sumar días a la fecha actual y obtener la fecha exacta proyectada.

# ── CELDA 6: Proyección de agotamiento ─────────────────────

def proyectar_agotamiento(df):
    """
    Calcula la fecha estimada en que cada producto se agotará
    si no se realiza ninguna orden de reabastecimiento.
    """
    df_proy = df.copy()
    hoy = datetime.now()
    
    # Fecha proyectada de agotamiento:
    # timedelta convierte días (float) a un objeto de duración de tiempo
    # apply() aplica una función fila por fila cuando la operación es compleja
    df_proy['fecha_agotamiento'] = df_proy.apply(
        lambda row: hoy + timedelta(days=float(row['dias_cobertura'])),
        axis=1
    )
    
    # Bandera: ¿el producto se agotará ANTES de que llegue el nuevo pedido?
    # True = hay riesgo real de quiebre de stock
    df_proy['riesgo_quiebre'] = (
        df_proy['dias_cobertura'] < df_proy['tiempo_entrega_dias']
    )
    
    # Calcular cantidad a pedir (lote económico simplificado)
    # Pedimos suficiente para llenar al máximo menos el stock actual
    df_proy['cantidad_a_pedir'] = np.maximum(
        df_proy['stock_maximo'] - df_proy['stock_actual'], 0
    )
    
    # Mostrar productos con riesgo de quiebre
    en_riesgo = df_proy[df_proy['riesgo_quiebre'] == True]
    print(f"⚠️  Productos con RIESGO DE QUIEBRE: {len(en_riesgo)}\n")
    
    for _, row in en_riesgo.iterrows():
        print(f"  🔴 {row['nombre']}")
        print(f"     Días cobertura : {row['dias_cobertura']:.1f} días")
        print(f"     Tiempo entrega : {row['tiempo_entrega_dias']} días")
        print(f"     Se agota el    : {row['fecha_agotamiento'].strftime('%d/%m/%Y')}")
        print(f"     Pedir urgente  : {row['cantidad_a_pedir']:.0f} unidades\n")
    
    return df_proy

df = proyectar_agotamiento(df)

Celda 7 — Clasificador de nivel de alerta por producto
Lógica condicional anidada para 4 niveles de criticidad
PythonLógica
○
¿Cómo se determinan los niveles de alerta?

El sistema clasifica cada producto en uno de cuatro niveles según múltiples condiciones:

🔴 CRÍTICO: stock actual ≤ stock mínimo, o stock actual ≤ punto de reorden Y hay riesgo de quiebre.
🟠 ADVERTENCIA: stock actual ≤ punto de reorden (tiempo de pedir, pero sin urgencia inmediata).
🟡 ATENCIÓN: stock actual ≤ 1.2× punto de reorden (margen estrecho).
🟢 NORMAL: stock suficiente, sin acción requerida.

Usamos numpy.select() que es más eficiente que múltiples if/elif para aplicar lógica condicional sobre un DataFrame completo.

# ── CELDA 7: Clasificador de alertas ────────────────────────

def clasificar_alertas(df):
    """
    Asigna un nivel de alerta a cada producto según su estado
    de inventario relativo al punto de reorden y stock mínimo.
    
    Niveles: CRÍTICO > ADVERTENCIA > ATENCIÓN > NORMAL
    """
    df_al = df.copy()
    
    # ── Definir condiciones (en orden de prioridad, más grave primero)
    # np.select evalúa condiciones en orden y asigna el primer True que encuentre
    condiciones = [
        # Condición CRÍTICO: agotado/mínimo O quiebre inminente
        (df_al['stock_actual'] <= df_al['stock_minimo']) |
        ((df_al['stock_actual'] <= df_al['punto_reorden']) &
         (df_al['riesgo_quiebre'] == True)),
        
        # Condición ADVERTENCIA: por debajo del punto de reorden
        df_al['stock_actual'] <= df_al['punto_reorden'],
        
        # Condición ATENCIÓN: en zona de margen ajustado (120% del ROP)
        df_al['stock_actual'] <= df_al['punto_reorden'] * 1.2
    ]
    
    # Valores a asignar según cada condición (mismo orden)
    valores = ['CRÍTICO', 'ADVERTENCIA', 'ATENCIÓN']
    
    # np.select: evalúa condiciones → asigna valor correspondiente
    # default='NORMAL' si ninguna condición es verdadera
    df_al['nivel_alerta'] = np.select(
        condiciones, valores, default='NORMAL'
    )
    
    # Icono visual para cada nivel (para reportes en consola)
    iconos = {'CRÍTICO':'🔴', 'ADVERTENCIA':'🟠',
              'ATENCIÓN':'🟡', 'NORMAL':'🟢'}
    df_al['icono_alerta'] = df_al['nivel_alerta'].map(iconos)
    
    # Prioridad numérica para ordenar alertas (menor = más urgente)
    prioridad = {'CRÍTICO':1, 'ADVERTENCIA':2, 'ATENCIÓN':3, 'NORMAL':4}
    df_al['prioridad'] = df_al['nivel_alerta'].map(prioridad)
    
    # Ordenar por prioridad (más crítico primero)

PLANTILLAS = {

┌─ 🔴 ALERTA CRÍTICA ──────────────────────────────┐
│ PRODUCTO: {r['nombre']}
│ Stock actual {r['stock_actual']} uds ≤ Mínimo {r['stock_minimo']} uds
│ SE AGOTA: {r['fecha_agotamiento'].strftime('%d/%m/%Y')}
│ PEDIR AHORA: {r['cantidad_a_pedir']:.0f} unidades
│ Para: almacén@empresa.com | responsable@empresa.com
└──────────────────────────────────────────────────┘""",

    'ADVERTENCIA': lambda r: f"""
┌─ 🟠 ADVERTENCIA ─────────────────────────────────┐
│ PRODUCTO: {r['nombre']}
│ Stock {r['stock_actual']} uds — por debajo del ROP ({r['punto_reorden']:.0f})
│ Cobertura restante: {r['dias_cobertura']:.1f} días
│ Programar pedido: {r['cantidad_a_pedir']:.0f} unidades
│ Para: almacén@empresa.com
└──────────────────────────────────────────────────┘""",

    'ATENCIÓN': lambda r: f"""
┌─ 🟡 ATENCIÓN ────────────────────────────────────┐
│ PRODUCTO: {r['nombre']} — margen ajustado
│ Stock {r['stock_actual']} uds — monitorear consumo
└──────────────────────────────────────────────────┘"""
}

def enviar_notificaciones(df):
    """
    Simula el envío de notificaciones automáticas por nivel de alerta.
    En producción, aquí llamarías a una API de email/SMS.
    """
    timestamp = datetime.now().strftime("%H:%M:%S")
    print(f"\n📡 [{timestamp}] Procesando notificaciones automáticas...\n")
    
    notifs_enviadas = 0
    
    # Iterar solo sobre productos con alerta activa (no NORMAL)
    alertas = df[df['nivel_alerta'] != 'NORMAL']
    
    for _, row in alertas.iterrows():
        nivel = row['nivel_alerta']
        
        # Obtener la plantilla correspondiente al nivel y generar el mensaje
        if nivel in PLANTILLAS:
            mensaje = PLANTILLAS[nivel](row)  # llamar la lambda con la fila
            print(mensaje)
            notifs_enviadas += 1
    
    print(f"\n✅ {notifs_enviadas} notificaciones procesadas.")
    print("   (En producción se enviarían por email/SMS/Slack)")

Lambda functions en Python

Una lambda es una función anónima de una sola expresión. lambda r: f"..." define una función sin nombre que recibe r (una fila del DataFrame) y devuelve un string formateado. Son ideales para funciones simples que se usan una sola vez, como las plantillas de mensajes aquí.

El diccionario PLANTILLAS actúa como un switch: en lugar de if nivel == 'CRÍTICO': ..., simplemente hacemos PLANTILLAS[nivel](row). Este patrón se llama dispatch table y es más limpio y extensible.

def crear_dashboard(df):
    """
    Genera un dashboard de 4 gráficas sobre el estado del inventario.
    Cada subplot muestra una dimensión diferente del análisis.
    """
    
    # ── Crear figura con 2×2 subplots (4 gráficas en una figura)
    # figsize controla el tamaño total en pulgadas (ancho × alto)
    fig, axes = plt.subplots(2, 2, figsize=(16, 12))
    fig.suptitle(
        f'Dashboard de Monitoreo de Inventario — {datetime.now().strftime("%d/%m/%Y")}',
        fontsize=14, fontweight='bold', color='white', y=1.01
    )
    
    # ── GRÁFICA 1 (axes[0,0]): Stock actual vs Punto de Reorden ──
    ax1 = axes[0, 0]
    
    # Asignar color a cada barra según nivel de alerta
    colores = [COLOR_ALERTA[n] for n in df['nivel_alerta']]
    
    # barh: barra horizontal. y=nombres, width=valores, color=lista de colores
    bars = ax1.barh(
        df['nombre'],
        df['stock_actual'],
        color=colores,
        alpha=0.85,
        edgecolor='white',
        linewidth=0.3
    )
    
    # Línea vertical punteada para el Punto de Reorden de cada producto
    # Como cada producto tiene su propio ROP, usamos scatter + vlines
    for i, (rop, nombre) in enumerate(zip(df['punto_reorden'], df['nombre'])):
        ax1.plot([rop, rop], [i-0.4, i+0.4],
                color='#58a6ff', linewidth=2, linestyle='--')
    
    ax1.set_title('Stock Actual vs Punto de Reorden', fontsize=11, pad=10)
    ax1.set_xlabel('Unidades')
    ax1.tick_params(axis='y', labelsize=8)
    
    # Leyenda manual usando parches de color
    leyenda = [mpatches.Patch(color=c, label=n)
               for n, c in COLOR_ALERTA.items()]
    ax1.legend(handles=leyenda, loc='lower right', fontsize=7)
    
    # ── GRÁFICA 2 (axes[0,1]): Distribución de alertas (pastel) ──
    ax2 = axes[0, 1]
    conteo = df['nivel_alerta'].value_counts()
    colores_pastel = [COLOR_ALERTA[n] for n in conteo.index]
    
    # pie chart: autopct muestra porcentaje en cada rebanada
    ax2.pie(
        conteo.values,
        labels=conteo.index,
        colors=colores_pastel,
        autopct='%1.0f%%',
        startangle=90,
        textprops={'fontsize': 9}
    )

    # ── GRÁFICA 3 (axes[1,0]): Valor de inventario por producto ──
    ax3 = axes[1, 0]

    colores3 = [COLOR_ALERTA[n] for n in df_sorted['nivel_alerta']]

    ax3.set_xlabel('Valor ($)')

    # ── GRÁFICA 4 (axes[1,1]): Días de cobertura por producto ────
    ax4 = axes[1, 1]
    colores4 = [COLOR_ALERTA[n] for n in df['nivel_alerta']]

    ax4.set_title('Días de Cobertura por Producto', fontsize=11, pad=10)
    ax4.set_ylabel('Días')
    ax4.legend(fontsize=8)
    
    # Ajuste automático de espaciado entre subplots
    plt.tight_layout()
    plt.show()
    print("✅ Dashboard generado correctamente")

def tabla_resumen_ejecutiva(df):
    """
    Genera una tabla resumen con las columnas más relevantes,
    formateada con colores según el nivel de alerta usando
    pandas Styler para visualización en Colab.
    """
    
    # Seleccionar y renombrar columnas para el reporte ejecutivo
    cols = {
        'icono_alerta'  : 'Alerta',
        'nombre'        : 'Producto',
        'clase_abc'     : 'ABC',
        'stock_actual'  : 'Stock',
        'punto_reorden' : 'ROP',
        'dias_cobertura': 'Días',
        'cantidad_a_pedir':'A Pedir',
        'nivel_alerta'  : 'Nivel'
    }
    
    tabla = df[list(cols.keys())].rename(columns=cols)
    
    # Función para colorear filas según nivel de alerta
    # Esta función se aplica fila por fila (axis=1)
    def colorear_fila(row):
        colores_bg = {
            'CRÍTICO'    : 'background-color: rgba(248,81,73,0.15)',
            'ADVERTENCIA': 'background-color: rgba(240,136,62,0.15)',
            'ATENCIÓN'   : 'background-color: rgba(210,153,34,0.15)',
            'NORMAL'     : ''
        }
        # Retorna lista de estilos CSS para cada celda de la fila
        estilo = colores_bg.get(row['Nivel'], '')
        return [estilo] * len(row)
    
    # Aplicar estilos: format() para decimales, apply() para colores de fila
    tabla_estilizada = tabla.style\
        .apply(colorear_fila, axis=1)\
        .format({
            'ROP'    : '{:.1f}',
            'Días'   : '{:.1f}',
            'A Pedir': '{:.0f}'
        })\
        .set_caption('Tabla Ejecutiva de Monitoreo de Inventario')\
        .hide(axis='index')
    
    return tabla_estilizada  # en Colab, devolver el Styler lo muestra automáticamente

# ── CELDA 12: Pipeline maestro del sistema ──────────────────
# ── Esta celda integra todas las funciones anteriores ────────

def ejecutar_monitoreo(datos, nivel_servicio=0.95, verbose=True):
    """
    Pipeline completo del Sistema de Monitoreo Automático de Inventario.
    
    Pasos del pipeline:
        1. Exploración inicial
        2. Cálculo de ROP y stock de seguridad
        3. Clasificación ABC
        4. Proyección de agotamiento
        5. Clasificación de alertas
        6. Generación de reporte
        7. Envío de notificaciones
        8. Generación de dashboard visual
        9. Retorno de tabla ejecutiva
    
    Parámetros:
        datos           : DataFrame del inventario
        nivel_servicio  : 0.90, 0.95 o 0.99 (default 95%)
        verbose         : si True, muestra output completo
    
    Retorna:
        DataFrame procesado con todas las métricas calculadas
    """
    
    # Mapear nivel de servicio a su valor Z de tabla normal
    tabla_z = {0.90: 1.282, 0.95: 1.645, 0.99: 2.326}
    z = tabla_z.get(nivel_servicio, 1.645)
    
    if verbose:
        print(f"\n🚀 Iniciando Sistema de Monitoreo...")
        print(f"   Nivel de servicio: {nivel_servicio*100:.0f}%  (Z={z})")
        print(f"   Productos a analizar: {len(datos)}\n")
    
    # ── Ejecutar pipeline secuencial ─────────────────────────
    # Cada función recibe el resultado de la anterior (encadenamiento)
    
    df_proc = datos.copy()
    
    if verbose: print("[1/6] Explorando inventario...")
    explorar_inventario(df_proc)
    
    if verbose: print("\n[2/6] Calculando puntos de reorden...")
    df_proc = calcular_rop(df_proc, z=z)
    
    if verbose: print("[3/6] Clasificando ABC...")
    df_proc = clasificar_abc(df_proc)
    
    if verbose: print("[4/6] Proyectando fechas de agotamiento...")
    df_proc = proyectar_agotamiento(df_proc)
    
    if verbose: print("[5/6] Clasificando niveles de alerta...")
    df_proc = clasificar_alertas(df_proc)
    
    if verbose:
        print("[6/6] Generando reporte y notificaciones...\n")
        generar_reporte_alertas(df_proc)
        enviar_notificaciones(df_proc)
    
    # Generar dashboard visual siempre
    crear_dashboard(df_proc)
    
    if verbose:
        print(f"\n✅ Sistema completado — {datetime.now().strftime('%d/%m/%Y %H:%M:%S')}")
    
    return df_proc
